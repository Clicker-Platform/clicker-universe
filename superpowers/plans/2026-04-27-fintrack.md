# Fintrack Module — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a per-site personal finance tracker module (Fintrack) into clicker-platform-v2 with wallet management, manual entries, transfers, dashboard analytics, AI receipt scanning, and advanced features (budget, goals, debts, recurring).

**Architecture:** Fresh module under `lib/modules/fintrack/` following the stocklens pattern. Firestore per-site under `sites/{siteId}/modules/fintrack/*`. AI scan borrows `gemini-scanner.ts` pattern from stocklens. No external chart library — CSS/SVG charts consistent with platform.

**Tech Stack:** Next.js App Router, Firebase Firestore (client SDK), Firebase Admin (server API routes), Google Gemini Vision, TypeScript, Tailwind CSS, Lucide icons, Sonner toasts.

**Working directory:** `main/clicker-platform-v2/`

**Branch:** Create `feature/fintrack` branch before starting.

```ts
import {

// ── Wallets ──────────────────────────────────────────────────────────────────

export async function getWallets(siteId: string): Promise<Wallet[]> {
  const q = query(collection(db, 'sites', siteId, FT_WALLETS), orderBy('createdAt'));
  const snap = await getDocs(q);
  return snap.docs.map(d => ({ id: d.id, ...d.data() } as Wallet));
}

export async function createWallet(siteId: string, data: Omit<Wallet, 'id' | 'createdAt'>): Promise<string> {
  const ref = await addDoc(collection(db, 'sites', siteId, FT_WALLETS), {
    ...data, createdAt: serverTimestamp(),
  });
  return ref.id;
}

export async function updateWallet(siteId: string, walletId: string, data: Partial<Omit<Wallet, 'id' | 'createdAt'>>): Promise<void> {
  await updateDoc(doc(db, 'sites', siteId, FT_WALLETS, walletId), data);
}

export async function deleteWallet(siteId: string, walletId: string): Promise<void> {
  await deleteDoc(doc(db, 'sites', siteId, FT_WALLETS, walletId));
}

export async function getWalletBalance(siteId: string, walletId: string): Promise<number> {
  const wallet = await getDoc(doc(db, 'sites', siteId, FT_WALLETS, walletId));
  if (!wallet.exists()) return 0;
  const saldoAwal = wallet.data().saldoAwal || 0;

  const q = query(collection(db, 'sites', siteId, FT_ENTRIES), where('walletId', '==', walletId));
  const snap = await getDocs(q);
  let balance = saldoAwal;
  snap.docs.forEach(d => {
    const entry = d.data();
    if (entry.jenis === 'pemasukan') balance += entry.jumlah;
    else balance -= entry.jumlah;
  });
  return balance;
}

// ── Entries ───────────────────────────────────────────────────────────────────

export async function getEntries(siteId: string, filters?: { bulan?: number; tahun?: number }): Promise<FintrackEntry[]> {
  const q = query(collection(db, 'sites', siteId, FT_ENTRIES), orderBy('tanggal', 'desc'), orderBy('createdAt', 'desc'));
  const snap = await getDocs(q);
  let entries = snap.docs.map(d => ({ id: d.id, ...d.data() } as FintrackEntry));
  if (filters?.bulan && filters?.tahun) {
    const prefix = `${filters.tahun}-${String(filters.bulan).padStart(2, '0')}`;
    entries = entries.filter(e => e.tanggal.startsWith(prefix));
  }
  return entries;
}

export async function createEntry(siteId: string, data: Omit<FintrackEntry, 'id' | 'createdAt'>): Promise<string> {
  const ref = await addDoc(collection(db, 'sites', siteId, FT_ENTRIES), {
    ...data, createdAt: serverTimestamp(),
  });
  return ref.id;
}

export async function updateEntry(siteId: string, entryId: string, data: Partial<Omit<FintrackEntry, 'id' | 'createdAt'>>): Promise<void> {

export async function deleteEntry(siteId: string, entryId: string): Promise<void> {
```

```tsx
'use client';

import { useState, useEffect, useCallback } from 'react';
import { toast } from 'sonner';
import { useSite } from '@/lib/site-context';
import { usePermission } from '@/components/admin/PermissionGuard';
import { Wallet } from '../types';
import {
  getWallets, createWallet, updateWallet, deleteWallet, getWalletBalance,
} from '../api';
import { WALLET_TYPE_SUGGESTIONS, WALLET_ICONS, WALLET_COLORS } from '../constants';
import { PlusCircle, Pencil, Trash2, Wallet as WalletIcon, ArrowLeftRight } from 'lucide-react';
import Link from 'next/link';

function formatRp(n: number) {
  return 'Rp ' + n.toLocaleString('id-ID');
}

export default function WalletVaultPage() {
  const { siteId } = useSite();
  const { isViewOnly } = usePermission();
  const [wallets, setWallets] = useState<Wallet[]>([]);
  const [balances, setBalances] = useState<Record<string, number>>({});
  const [loading, setLoading] = useState(true);
  const [showForm, setShowForm] = useState(false);
  const [editing, setEditing] = useState<Wallet | null>(null);

  const [form, setForm] = useState({ nama: '', tipe: '', icon: '🏦', warna: WALLET_COLORS[0], saldoAwal: '' });

  const load = useCallback(async () => {
    if (!siteId) return;
    setLoading(true);
    const ws = await getWallets(siteId);
    setWallets(ws);
    const bals: Record<string, number> = {};
    await Promise.all(ws.map(async w => { bals[w.id] = await getWalletBalance(siteId, w.id); }));
    setBalances(bals);
    setLoading(false);
  }, [siteId]);

  useEffect(() => { load(); }, [load]);

  function openNew() { setForm({ nama: '', tipe: '', icon: '🏦', warna: WALLET_COLORS[0], saldoAwal: '' }); setEditing(null); setShowForm(true); }
  function openEdit(w: Wallet) { setForm({ nama: w.nama, tipe: w.tipe, icon: w.icon, warna: w.warna, saldoAwal: String(w.saldoAwal) }); setEditing(w); setShowForm(true); }

  async function handleSave() {
    if (!form.nama.trim()) return toast.error('Nama dompet wajib diisi');
    const data = { nama: form.nama.trim(), tipe: form.tipe, icon: form.icon, warna: form.warna, saldoAwal: Number(form.saldoAwal) || 0 };
    if (editing) {
      await updateWallet(siteId, editing.id, data);
      toast.success('Dompet diperbarui');
    } else {
      await createWallet(siteId, data);
      toast.success('Dompet ditambahkan');
    }
    setShowForm(false);
    load();
  }

  async function handleDelete(w: Wallet) {
    if (!confirm(`Hapus dompet "${w.nama}"?`)) return;
    await deleteWallet(siteId, w.id);
    toast.success('Dompet dihapus');
    load();
  }

  const totalSaldo = Object.values(balances).reduce((a, b) => a + b, 0);

  return (
    <div className="p-6 max-w-3xl mx-auto space-y-6">
      <div className="flex items-center justify-between">
        <div>
          <h1 className="text-2xl font-bold dark:text-white flex items-center gap-2"><WalletIcon className="w-6 h-6" /> Vault Dompet</h1>
          <p className="text-sm text-gray-500 dark:text-gray-400">Total semua: {formatRp(totalSaldo)}</p>
        </div>
        {!isViewOnly && (
          <button onClick={openNew} className="flex items-center gap-2 px-4 py-2 bg-amber-500 hover:bg-amber-600 text-white rounded-lg text-sm font-medium">
            <PlusCircle className="w-4 h-4" /> Tambah Dompet
          </button>
        )}
      </div>

      {loading ? (
        <p className="text-gray-500 dark:text-gray-400">Memuat...</p>
      ) : wallets.length === 0 ? (
        <div className="text-center py-16 text-gray-400">
          <WalletIcon className="w-12 h-12 mx-auto mb-3 opacity-30" />
          <p>Belum ada dompet. Tambahkan yang pertama!</p>
        </div>
      ) : (
        <div className="space-y-3">
          {wallets.map(w => (
            <div key={w.id} className="flex items-center justify-between p-4 bg-white dark:bg-neutral-800 rounded-xl border border-gray-200 dark:border-neutral-700">
              <div className="flex items-center gap-3">
                <span className="text-2xl">{w.icon}</span>
                <div>
                  <p className="font-semibold dark:text-white">{w.nama}</p>
                  <p className="text-xs text-gray-500 dark:text-gray-400">{w.tipe || 'Umum'}</p>
                </div>
              </div>
              <div className="flex items-center gap-4">
                <p className="font-bold text-lg dark:text-white">{formatRp(balances[w.id] ?? w.saldoAwal)}</p>
                {!isViewOnly && (
                  <div className="flex gap-1">
                    <button onClick={() => openEdit(w)} className="p-2 hover:bg-gray-100 dark:hover:bg-neutral-700 rounded-lg"><Pencil className="w-4 h-4 text-gray-500" /></button>
                    <button onClick={() => handleDelete(w)} className="p-2 hover:bg-red-50 dark:hover:bg-red-900/20 rounded-lg"><Trash2 className="w-4 h-4 text-red-500" /></button>
                  </div>
                )}
              </div>
            </div>
          ))}
        </div>
      )}

      {/* Form Modal */}
      {showForm && (
        <div className="fixed inset-0 bg-black/50 z-50 flex items-center justify-center p-4">
          <div className="bg-white dark:bg-neutral-900 rounded-2xl p-6 w-full max-w-md space-y-4">
            <h2 className="text-lg font-bold dark:text-white">{editing ? 'Edit Dompet' : 'Tambah Dompet'}</h2>
            <input className="w-full border dark:border-neutral-700 dark:bg-neutral-800 dark:text-white rounded-lg px-3 py-2 text-sm" placeholder="Nama dompet*" value={form.nama} onChange={e => setForm(p => ({ ...p, nama: e.target.value }))} />
            <input className="w-full border dark:border-neutral-700 dark:bg-neutral-800 dark:text-white rounded-lg px-3 py-2 text-sm" placeholder="Tipe (Bank, E-Wallet, Cash...)" value={form.tipe} onChange={e => setForm(p => ({ ...p, tipe: e.target.value }))} list="tipe-list" />
```

```tsx
function formatRp(n: number) { return 'Rp ' + n.toLocaleString('id-ID'); }
function shortRp(n: number) {
  if (Math.abs(n) >= 1_000_000_000) return (n / 1_000_000_000).toFixed(1) + 'M';
  if (Math.abs(n) >= 1_000_000) return (n / 1_000_000).toFixed(1) + 'jt';
  if (Math.abs(n) >= 1_000) return (n / 1_000).toFixed(0) + 'rb';
  return String(n);
}

export default function DashboardPage() {
  const { siteId } = useSite();
  const now = new Date();
  const [bulan, setBulan] = useState(now.getMonth() + 1);
  const [tahun, setTahun] = useState(now.getFullYear());
  const [wallets, setWallets] = useState<Wallet[]>([]);
  const [entries, setEntries] = useState<FintrackEntry[]>([]);
  const [totalSaldo, setTotalSaldo] = useState(0);
  const [loading, setLoading] = useState(true);

  const load = useCallback(async () => {
    if (!siteId) return;
    setLoading(true);
    const [ws, es] = await Promise.all([getWallets(siteId), getEntries(siteId, { bulan, tahun })]);
    setWallets(ws);
    setEntries(es);
    const bals = await Promise.all(ws.map(w => getWalletBalance(siteId, w.id)));

  const pengeluaran = entries.filter(e => e.jenis === 'pengeluaran').reduce((a, e) => a + e.jumlah, 0);
  const cashflow = pemasukan - pengeluaran;
  const daysInMonth = new Date(tahun, bulan, 0).getDate();
  const rataHarian = Math.abs(cashflow) / daysInMonth;

  // Category distribution
  const catMap: Record<string, number> = {};
  entries.filter(e => e.jenis === 'pengeluaran').forEach(e => { catMap[e.kategori] = (catMap[e.kategori] || 0) + e.jumlah; });
  const catEntries = Object.entries(catMap).sort(([, a], [, b]) => b - a).slice(0, 5);

  for (let d = 1; d <= daysInMonth; d++) dayBuckets[d] = { masuk: 0, keluar: 0 };

  });
  const chartDays = Object.entries(dayBuckets).map(([d, v]) => ({ day: Number(d), ...v }));
  const maxVal = Math.max(...chartDays.map(d => Math.max(d.masuk, d.keluar)), 1);
```

```ts
import { GoogleGenerativeAI } from '@google/generative-ai';
import { adminDb } from '@/lib/firebase-admin';
import { logger } from '@/lib/logger';
import { FT_CONFIG } from '../constants';

async function getApiKey(siteId: string): Promise<string | null> {
  try {
    const snap = await adminDb.doc(`sites/${siteId}/${FT_CONFIG}`).get();
    if (snap.exists) return snap.data()?.apiKey || null;
    return process.env.GEMINI_API_KEY || null;
  } catch (e) {
    logger.error('fintrack.apikey.fetch.failed', { siteId, error: e });
    return process.env.GEMINI_API_KEY || null;
  }
}

const RECEIPT_PROMPT = `Kamu adalah asisten scan struk belanja.
Analisa gambar struk/receipt dan ekstrak informasi berikut.

Kembalikan HANYA JSON valid, tidak ada teks lain:
{
  "total_bayar": 60300,
  "tanggal": "2026-03-04",
  "item_list": ["nama item 1", "nama item 2"],
  "toko": "Nama toko jika ada"
}

Rules:
- total_bayar: angka saja tanpa simbol, dalam rupiah
- tanggal: format YYYY-MM-DD, gunakan hari ini jika tidak jelas
- item_list: array nama item yang dibeli
- toko: nama merchant/toko, kosong string jika tidak ada`;

export async function scanReceipt(siteId: string, imageBase64: string, mimeType: string) {
  const apiKey = await getApiKey(siteId);
  if (!apiKey) throw new Error('Gemini API Key belum dikonfigurasi. Silakan atur di Settings.');

  const ai = new GoogleGenerativeAI(apiKey);
  const model = ai.getGenerativeModel({ model: 'gemini-2.0-flash' });

  const result = await model.generateContent([
    RECEIPT_PROMPT,
    { inlineData: { data: imageBase64, mimeType } },
  ]);

  const text = result.response.text();
  const jsonStr = text.replace(/```json\n?/g, '').replace(/```\n?/g, '').trim();

  let parsed: Record<string, unknown>;
  try {
    parsed = JSON.parse(jsonStr) as Record<string, unknown>;
  } catch {
    logger.error('fintrack.receipt.parse.failed', { siteId, raw: text });
    return { total_bayar: 0, tanggal: new Date().toISOString().split('T')[0], item_list: [], toko: '' };
  }

  return {
    total_bayar: Number(parsed.total_bayar) || 0,
    tanggal: (parsed.tanggal as string) || new Date().toISOString().split('T')[0],
    item_list: (parsed.item_list as string[]) || [],
    toko: (parsed.toko as string) || '',
  };
}
```

```ts
export async function POST(req: NextRequest) {
  try {
    const formData = await req.formData();
    const siteId = formData.get('siteId') as string;
    const file = formData.get('image') as File;

    if (!siteId || !file) return NextResponse.json({ error: 'siteId and image required' }, { status: 400 });

    const buffer = await file.arrayBuffer();
    const base64 = Buffer.from(buffer).toString('base64');
    const mimeType = file.type || 'image/jpeg';

    // Upload to Firebase Storage
    const storage = getStorage(adminApp);
    const fileName = `fintrack/${siteId}/struk-${Date.now()}.${file.type.split('/')[1] || 'jpg'}`;
    const storageRef = ref(storage, fileName);
    await uploadBytes(storageRef, Buffer.from(buffer), { contentType: mimeType });
    const buktiUrl = await getDownloadURL(storageRef);

    const scan = await scanReceipt(siteId, base64, mimeType);

    return NextResponse.json({
      jumlah: scan.total_bayar,
      tanggal: scan.tanggal,
      catatan: scan.item_list.join(', '),
      toko: scan.toko,
      buktiUrl,
    });
  } catch (error: unknown) {
    logger.error('fintrack.scan-struk.route.failed', { error });
    const message = error instanceof Error ? error.message : 'Unknown error';
    return NextResponse.json({ error: message }, { status: 500 });
  }
```

```ts
export const dynamic = 'force-dynamic';

export async function GET(req: NextRequest) {

  try {
    const snap = await adminDb.doc(`sites/${siteId}/${FT_CONFIG}`).get();
    return NextResponse.json({ hasKey: snap.exists && !!snap.data()?.apiKey });
  } catch (error: unknown) {
    logger.error('fintrack.settings.get.failed', { error });
    return NextResponse.json({ error: 'Failed' }, { status: 500 });
  }
}

export async function POST(req: NextRequest) {
  try {
    const { siteId, apiKey } = await req.json();
    if (!siteId) return NextResponse.json({ error: 'siteId required' }, { status: 400 });
    await adminDb.doc(`sites/${siteId}/${FT_CONFIG}`).set({ apiKey: apiKey || null, updatedAt: Date.now() }, { merge: true });
    return NextResponse.json({ success: true });
  } catch (error: unknown) {
    logger.error('fintrack.settings.post.failed', { error });
    return NextResponse.json({ error: 'Failed' }, { status: 500 });
  }
```

```tsx
    </div>
  );
}

// ── Debts Tab ─────────────────────────────────────────────────────────────────

function DebtsTab({ siteId, isViewOnly }: { siteId: string; isViewOnly: boolean }) {
  const [items, setItems] = useState<Debt[]>([]);
  const [showForm, setShowForm] = useState(false);
  const [form, setForm] = useState({ nama: '', jenis: 'hutang' as const, jumlah: '', catatan: '', deadline: '' });

  const load = useCallback(async () => { setItems(await getDebts(siteId)); }, [siteId]);
  useEffect(() => { load(); }, [load]);

  async function handleSave() {
    if (!form.nama || !form.jumlah) return toast.error('Nama dan jumlah wajib diisi');
    await createDebt(siteId, { nama: form.nama, jenis: form.jenis, jumlah: Number(form.jumlah), lunas: false, catatan: form.catatan, deadline: form.deadline || undefined });
    toast.success('Hutang/piutang ditambahkan'); setShowForm(false); load();
  }

  return (
    <div className="space-y-3">
      {!isViewOnly && <button onClick={() => setShowForm(true)} className="flex items-center gap-2 px-4 py-2 bg-amber-500 hover:bg-amber-600 text-white rounded-lg text-sm font-medium"><Plus className="w-4 h-4" /> Tambah</button>}
      {items.map(d => (
        <div key={d.id} className="bg-white dark:bg-neutral-800 rounded-xl p-4 border border-gray-100 dark:border-neutral-700">
          <div className="flex justify-between items-center">
            <div>
              <p className="font-medium dark:text-white">{d.nama}</p>
              <p className={`text-xs font-medium ${d.jenis === 'hutang' ? 'text-red-500' : 'text-green-600 dark:text-green-400'}`}>{d.jenis === 'hutang' ? 'Hutang Saya' : 'Piutang Saya'} · {formatRp(d.jumlah)}</p>
            </div>
            <div className="flex items-center gap-2">
              {!d.lunas && !isViewOnly && (
                <button onClick={async () => { await updateDebt(siteId, d.id, { lunas: true }); load(); }} className="text-xs px-2 py-1 bg-green-100 dark:bg-green-900/30 text-green-700 dark:text-green-400 rounded-lg">Lunas</button>
              )}
              {d.lunas && <CheckCircle2 className="w-4 h-4 text-green-500" />}
              {!isViewOnly && <button onClick={async () => { await deleteDebt(siteId, d.id); load(); }}><Trash2 className="w-4 h-4 text-red-400" /></button>}
            </div>
          </div>
        </div>
      ))}
      {items.length === 0 && <p className="text-center py-8 text-gray-400 text-sm">Belum ada catatan hutang/piutang.</p>}

      {showForm && (
        <div className="fixed inset-0 bg-black/50 z-50 flex items-center justify-center p-4">
          <div className="bg-white dark:bg-neutral-900 rounded-2xl p-6 w-full max-w-sm space-y-3">
            <h3 className="font-bold dark:text-white">Tambah Hutang / Piutang</h3>
            <input className="w-full border dark:border-neutral-700 dark:bg-neutral-800 dark:text-white rounded-lg px-3 py-2 text-sm" placeholder="Nama orang*" value={form.nama} onChange={e => setForm(p => ({ ...p, nama: e.target.value }))} />
            <div className="flex gap-2">
              {(['hutang', 'piutang'] as const).map(j => (
                <button key={j} onClick={() => setForm(p => ({ ...p, jenis: j }))} className={`flex-1 py-2 rounded-lg text-sm border-2 ${form.jenis === j ? 'border-amber-500 bg-amber-50 dark:bg-amber-900/20 text-amber-700 dark:text-amber-400' : 'border-gray-200 dark:border-neutral-700 dark:text-white'}`}>
                  {j === 'hutang' ? 'Hutang Saya' : 'Piutang Saya'}
                </button>
              ))}
            </div>
            <input type="number" className="w-full border dark:border-neutral-700 dark:bg-neutral-800 dark:text-white rounded-lg px-3 py-2 text-sm" placeholder="Jumlah (Rp)*" value={form.jumlah} onChange={e => setForm(p => ({ ...p, jumlah: e.target.value }))} />
            <input className="w-full border dark:border-neutral-700 dark:bg-neutral-800 dark:text-white rounded-lg px-3 py-2 text-sm" placeholder="Catatan" value={form.catatan} onChange={e => setForm(p => ({ ...p, catatan: e.target.value }))} />
            <div className="flex gap-2">
              <button onClick={() => setShowForm(false)} className="flex-1 px-3 py-2 border dark:border-neutral-700 dark:text-white rounded-lg text-sm">Batal</button>
              <button onClick={handleSave} className="flex-1 px-3 py-2 bg-amber-500 text-white rounded-lg text-sm">Simpan</button>
            </div>
          </div>
        </div>
      )}
    </div>
  );
}

function RecurringTab({ siteId, isViewOnly }: { siteId: string; isViewOnly: boolean }) {
  const [items, setItems] = useState<Recurring[]>([]);
  const [showForm, setShowForm] = useState(false);

    toast.success('Transaksi berulang ditambahkan'); setShowForm(false); load();
  }

  return (
    <div className="space-y-3">
      {!isViewOnly && <button onClick={() => setShowForm(true)} className="flex items-center gap-2 px-4 py-2 bg-amber-500 hover:bg-amber-600 text-white rounded-lg text-sm font-medium"><Plus className="w-4 h-4" /> Tambah</button>}

              <p className="font-medium dark:text-white">{r.judul}</p>
              <p className="text-xs text-gray-500 dark:text-gray-400">{r.frekuensi} · {formatRp(r.jumlah)} · {r.aktif ? 'Aktif' : 'Nonaktif'}</p>
            </div>
            <div className="flex items-center gap-2">
              {!isViewOnly && (
                <>
                  <button onClick={async () => { await updateRecurring(siteId, r.id, { aktif: !r.aktif }); load(); }} className={`text-xs px-2 py-1 rounded-lg ${r.aktif ?           </div>
        </div>

        <div className="fixed inset-0 bg-black/50 z-50 flex items-center justify-center p-4">

            <input type="number" className="w-full border dark:border-neutral-700 dark:bg-neutral-800 dark:text-white rounded-lg px-3 py-2 text-sm" placeholder="Jumlah (Rp)*" value={form.jumlah} onChange={e => setForm(p => ({ ...p, jumlah: e.target.value }))} />
            <select className="w-full border dark:border-neutral-700 dark:bg-neutral-800 dark:text-white rounded-lg px-3 py-2 text-sm" value={form.frekuensi} onChange={e => setForm(p => ({ ...p, frekuensi: e.target.value as typeof form.frekuensi }))}>
              {['harian','mingguan','bulanan','tahunan'].map(f => <option key={f} value={f}>{f.charAt(0).toUpperCase() + f.slice(1)}</option>)}
            </select>
            <input type="date" className="w-full border dark:border-neutral-700 dark:bg-neutral-800 dark:text-white rounded-lg px-3 py-2 text-sm" value={form.tglMulai} onChange={e => setForm(p => ({ ...p, tglMulai: e.target.value }))} />
            <div className="flex gap-2">
              <button onClick={() => setShowForm(false)} className="flex-1 px-3 py-2 border dark:border-neutral-700 dark:text-white rounded-lg text-sm">Batal</button>
              <button onClick={handleSave} className="flex-1 px-3 py-2 bg-amber-500 text-white rounded-lg text-sm">Simpan</button>
            </div>
          </div>
        </div>
      )}
    </div>
  );
}
```
